    // Mostrar/ocultar controles arcade según modo
    const arcadeControls = document.getElementById('arcadeControls');
    const controlModeSel = document.getElementById('controlMode');
    controlModeSel.onchange = () => {
      if (controlModeSel.value === "buttons") {
        arcadeControls.style.display = "flex";
        // desactiva gestos táctiles
      } else {
        arcadeControls.style.display = "none";
        // activa gestos táctiles
        setupGestures();
      }
    };
    function setupGestures() {
      // Ejemplo básico: Swipe para mover, tap para rotar
      let startX, startY;
      const board = document.getElementById("board");
      board.ontouchstart = function(e) {
        if (e.touches.length === 1) {
          startX = e.touches[0].clientX;
          startY = e.touches[0].clientY;
        }
      };
      board.ontouchend = function(e) {
        if (!startX || !startY) return;
        const dx = e.changedTouches[0].clientX - startX;
        const dy = e.changedTouches[0].clientY - startY;
        if (Math.abs(dx) > Math.abs(dy)) {
          if (dx > 40) { /* mover derecha */ }
          else if (dx < -40) { /* mover izquierda */ }
        } else {
          if (dy > 40) { /* bajar rápido */ }
          else if (dy < -40) { /* rotar pieza */ }
        }
        startX = startY = null;
      };
    }
    // Inicializa el modo visual y controles
    visualModeSel.value = "bn";
    controlModeSel.value = "buttons";
    arcadeControls.style.display = "flex";
  </script>
